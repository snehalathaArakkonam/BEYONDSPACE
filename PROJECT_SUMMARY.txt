```html
<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">

    <title>[PAGE TITLE] • BEYONDSPACE</title>
    <link rel="stylesheet" href="css/main.css">
</head>
<body>
    <canvas id="particles" class="particle-canvas"></canvas>
    
    <nav class="navbar"><!-- COPY FULL NAVBAR FROM index.html --></nav>
    
    <header class="page-header">
        <h1 data-en="[TITLE]" data-te="[TELUGU TITLE]">[TITLE]</h1>
        <p data-en="[SUBTITLE]" data-te="[TELUGU SUBTITLE]">[SUBTITLE]</p>
    </header>
    
    <main class="[page]-main">
        <!-- PAGE SPECIFIC CONTENT -->
    </main>
    
    <footer class="footer"><!-- COPY FULL FOOTER FROM index.html --></footer>
    <button id="back-to-top" class="back-to-top">🚀</button>
    
    <script src="js/particles.js"></script>
    <script src="js/main.js"></script>
    <!-- PAGE SPECIFIC SCRIPTS -->
</body>
</html>
```
